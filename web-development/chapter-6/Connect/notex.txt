-- A 'middleware' is a Javascript function
-- Middlewares commonly follows a simple convention in order to provide configurations to capabilities to developers:
    using a funtion that returns another function (a closure).

    ex: function setup(option) {
        // configs here

        return function (req, res, next) { 
            // middleware logic
        }
    }
